UndefinedMethodFix

IsUndefinedMethod

  • false if method can be resolved

    '''
describe "Something"{
  String x
  fact x.toString
}
'''.hasMissingMethod => false

  • false if method receiver cannot be resolved

    '''
describe "Something"{
  fact x.toString
}
'''.hasMissingMethod => false

  • false if method receiver has unknown type

    '''
describe "Something"{
  Unknown x
  fact x.unresolved
}
'''.hasMissingMethod => false

  • false if method receiver is literal

    '''
describe "Something"{
  fact 9.toString
}
'''.hasMissingMethod => false

  • true if method cannot be resolved

    '''
describe "Something"{
  String x
  fact x.unresolved
}
'''.hasMissingMethod => true

  • true if method exists with different parameter count

    '''
describe "Something"{
  String x
  fact x.toString("hello")
}
'''.hasMissingMethod => true

  • true if method exists with different parameter types

    '''
describe "Something"{
  String x
  fact x.concat(4)
}
'''.hasMissingMethod => true

CallsReadOnlyType

  • true if java element is read-only

    javaElementProvider.setReadOnly
'''
describe "Something"{
  String x
  fact x.unresolved
}
'''.receiverIsReadOnly => true

  • false if java element is changeable

    '''
describe "Something"{
  SomethingSpec x
  fact x.unresolved

'''.receiverIsReadOnly => false
