SpecResolver

  • resolves referenced specs

    parseSuite('''
  package demo
  import demo.*
  #My Suite
  
  - "My Spec"
  - "My Feature"
''')

resolvedSpecs(firstSuite) => set("MySpecSpec", "MyFeatureFeature")

package demo
